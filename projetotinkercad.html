        void setup()
        {
          pinMode(12, OUTPUT);
          pinMode(8, OUTPUT);
          pinMode(4, OUTPUT);
        }
        
        void loop()
        {
          digitalWrite(12, HIGH);
          delay(5000); // Wait for 5000 millisecond(s)
          digitalWrite(12, LOW);
          delay(2000); // Wait for 2000 millisecond(s)
          digitalWrite(8, HIGH);
          delay(5000); // Wait for 5000 millisecond(s)
          digitalWrite(8, LOW);
          delay(2000); // Wait for 2000 millisecond(s)
          digitalWrite(4, HIGH);
          delay(5000); // Wait for 5000 millisecond(s)
          digitalWrite(4, LOW);
          delay(1); // Wait for 1 millisecond(s)
        }
